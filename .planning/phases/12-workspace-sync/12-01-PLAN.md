  name: string,

To:

```typescript
export async function syncWorkspace(
  name: string,
  opts: { strategy?: "rebase" | "merge"; bestEffort?: boolean },
  onProgress?: ProgressCallback,
  onSyncProgress?: (update: SyncRow) => void
): Promise<SyncResult>
```

**Step 3: Replace fetch failure swallowing** (currently line ~707). Change the fetch block from:

```typescript
await Promise.all(
  repoInfos
    .filter(({ repo }) => existsSync(repo.task_path))
    .map(({ repo }) => fetchOrigin(repo.task_path).catch(() => {}))
)
```

To:

```typescript
const fetchFailures = new Map<string, string>()
await Promise.all(
  repoInfos
    .filter(({ repo }) => existsSync(repo.task_path))
    .map(async ({ repo }) => {
      onSyncProgress?.({ repo: repo.name, status: "fetching", detail: "", conflicts: [] })
      try {
        await fetchOrigin(repo.task_path)
      } catch (e) {
        const msg = e instanceof Error ? e.message : "fetch failed"
        const detail = msg.includes("timeout") ? "fetch failed (timeout)" : "fetch failed"
        fetchFailures.set(repo.name, detail)
        onSyncProgress?.({ repo: repo.name, status: "failed", detail, conflicts: [] })
      }
    })
)
```

**Step 4: Skip fetch-failed repos in the rebase loop** (line ~738). After the `if (!existsSync(repo.task_path))` check, add:

```typescript
if (fetchFailures.has(repo.name)) {
  skipped.push({ repo: repo.name, reason: fetchFailures.get(repo.name)! })
  onProgress?.(`failed  ${repo.name}  ${fetchFailures.get(repo.name)}`)
  continue
}
```

**Step 5: Add onSyncProgress calls in the rebase loop** for status transitions:

Before the conflict check: `onSyncProgress?.({ repo: repo.name, status: "rebasing", detail: "", conflicts: [] })`

On conflict skip (after `skipped.push`): `onSyncProgress?.({ repo: repo.name, status: "skipped", detail: \`conflict: ${c.files[0]}\`, conflicts: c.files.slice(1) })`

On rebase/merge failure (after `skipped.push`): `onSyncProgress?.({ repo: repo.name, status: "failed", detail: result.error ?? "sync failed", conflicts: [] })`

On success (after `synced.push`): `onSyncProgress?.({ repo: repo.name, status: "synced", detail: \`+${commitsBefore} commits\`, conflicts: [] })`
