   ```typescript
     ensureDir(dirname(path))
     const tmpPath = `${path}.tmp`
     const content = stringify(data)
     const fd = openSync(tmpPath, "w")
     try {
       writeSync(fd, content, 0, "utf-8")
       fsyncSync(fd)
     } finally {
       closeSync(fd)
     }
     renameSync(tmpPath, path)
   }
   ```
