     op: "copy" | "symlink",
     entries: string[],
     sourceBaseDir: string,
     destDir: string

         const matches = expandGlob(entry, sourceBaseDir)

           warnings.push(`Glob '${entry}' matched no files in ${sourceBaseDir}`)
           continue

         for (const match of matches) {

           const dst = join(destDir, match)

           if (!r.ok) return r

```typescript
/**
 * Create a directory tree from a flat map of relative paths to contents.
 * Paths ending with "/" create directories. Others create files.
 * Example: makeFileTree(base, { "a/b.txt": "hello", "c/": "" })
 */
export function makeFileTree(base: string, entries: Record<string, string>) {
  for (const [rel, content] of Object.entries(entries)) {
    if (rel.endsWith("/")) {
      mkdirSync(join(base, rel), { recursive: true })
    } else {
      const full = join(base, rel)
      mkdirSync(dirname(full), { recursive: true })
      writeFileSync(full, content)
    }
  }
}
```
