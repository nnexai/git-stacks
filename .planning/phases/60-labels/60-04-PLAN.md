```ts
const filteredEntries = createMemo(() => {
  const f = tabFilter.workspaces[0]().toLowerCase()
  if (!f) return entries()
  return entries().filter((e) => {
    const name = e.workspace.name.toLowerCase()
    const labels = e.workspace.labels ?? []
    // label: prefix — match only against labels
    if (f.startsWith("label:")) {
      const labelTerm = f.slice(6)
      if (!labelTerm) return true
      return labels.some(l => l.toLowerCase().includes(labelTerm))
    }
    // Default: match name OR any label
    return name.includes(f) || labels.some(l => l.toLowerCase().includes(f))
  })
})
```

Key behavior:
- `/backend` matches workspace name containing "backend" OR any label containing "backend"
- `/label:sprint:14` matches only labels containing "sprint:14"
- `/label:` with nothing after shows all (no filter)
- Filter is case-insensitive for both name and label matching (standard for TUI search)

2. Add a `groupedEntries` memo after `filteredEntries`. This produces a flat array of items that can be either group headers or workspace entries:

```ts
type GroupedItem =
  | { kind: "header"; label: string }
  | { kind: "entry"; entry: WorkspaceEntry; originalIndex: number }

const groupedEntries = createMemo((): GroupedItem[] => {
  if (!groupedByLabel() || tab() !== "workspaces") return []
  const all = filteredEntries()
  const labelMap = new Map<string, { entry: WorkspaceEntry; originalIndex: number }[]>()
  const unlabeled: { entry: WorkspaceEntry; originalIndex: number }[] = []

  all.forEach((entry, i) => {
    const labels = entry.workspace.labels ?? []
    if (labels.length === 0) {
      unlabeled.push({ entry, originalIndex: i })
    } else {
      for (const label of labels) {
        if (!labelMap.has(label)) labelMap.set(label, [])
        labelMap.get(label)!.push({ entry, originalIndex: i })
      }
    }
  })

  const items: GroupedItem[] = []
  // Sort labels alphabetically
  const sortedLabels = [...labelMap.keys()].sort()
  for (const label of sortedLabels) {
    items.push({ kind: "header", label })
    for (const item of labelMap.get(label)!) {
      items.push({ kind: "entry", ...item })
    }
  }
  if (unlabeled.length > 0) {
    items.push({ kind: "header", label: "[unlabeled]" })
    for (const item of unlabeled) {
      items.push({ kind: "entry", ...item })
    }
  }
  return items
})
```

```ts
// For grouped view: build a flat list of navigable (non-header) items
const navigableItems = createMemo(() => {
  if (!props.isGrouped) return []
  return props.grouped
    .map((item, i) => ({ item, groupIdx: i }))
    .filter(({ item }) => item.kind === "entry")
})

// Map cursor (navigable index) to grouped array index
const cursorToGroupIdx = createMemo(() => {
  const nav = navigableItems()
  if (props.cursor >= 0 && props.cursor < nav.length) {
    return nav[props.cursor].groupIdx
  }
  return -1
})
```

3. Replace the render section to handle both modes. Wrap the existing flat rendering in a `Show when={!props.isGrouped}` and add grouped rendering:

```tsx
return (
  <box flexDirection="column">
    <Show when={!props.isGrouped}>
      {/* Existing flat list rendering — unchanged */}
      <Show
        when={props.entries.length > 0}
        fallback={<text fg="gray">  No workspaces found. Run `git-stacks new` to create one.</text>}
      >
        <For each={visibleEntries()}>
          {(entry) => {
            const realIndex = () => props.entries.indexOf(entry)
            return (
              <WorkspaceRow
                entry={entry}
                focused={realIndex() === props.cursor}
                selected={props.selected.has(realIndex())}
                messages={props.allMessages.get(entry.workspace.name) ?? []}
                tick={props.tick}
              />
            )
          }}
        </For>
        <Show when={props.entries.length > viewportHeight()}>
          <text fg="gray">
            {`  ${scrollOffset() + 1}-${Math.min(scrollOffset() + viewportHeight(), props.entries.length)} of ${props.entries.length}`}
          </text>
        </Show>
      </Show>
    </Show>

    <Show when={props.isGrouped}>
      <Show
        when={props.grouped.length > 0}
        fallback={<text fg="gray">  No workspaces found.</text>}
      >
        <For each={groupedVisible()}>
          {(item, groupIdx) => {
            if (item.kind === "header") {
              return (
                <box height={1}>
                  <text fg="yellow" bold>{`  ${item.label}`}</text>
                </box>
              )
            }
            // Find this entry's cursor position among navigable items
            const navIdx = () => {
              const nav = navigableItems()
              return nav.findIndex(n => n.groupIdx === groupedScrollOffset() + groupIdx())
            }
            return (
              <WorkspaceRow
                entry={item.entry}
                focused={navIdx() === props.cursor}
                selected={props.selected.has(item.originalIndex)}
                messages={props.allMessages.get(item.entry.workspace.name) ?? []}
                tick={props.tick}
              />
            )
          }}
        </For>
        <Show when={props.grouped.length > viewportHeight()}>
          <text fg="gray">
            {`  Showing ${navigableItems().length} workspaces in ${props.grouped.filter(g => g.kind === "header").length} groups`}
          </text>
        </Show>
      </Show>
    </Show>
  </box>
)
```

4. Add scroll memos for grouped view:

```ts
const groupedScrollOffset = createMemo(() => {
  if (!props.isGrouped) return 0
  const targetGroupIdx = cursorToGroupIdx()
  if (targetGroupIdx < 0) return 0
  const vh = viewportHeight()
  if (targetGroupIdx < vh) return 0
  return Math.min(targetGroupIdx - vh + 1, Math.max(0, props.grouped.length - vh))
})

const groupedVisible = createMemo(() => {
  if (!props.isGrouped) return []
  return props.grouped.slice(groupedScrollOffset(), groupedScrollOffset() + viewportHeight())
})
```
