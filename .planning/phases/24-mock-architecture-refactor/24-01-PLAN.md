**lifecycle.ts refactor (per D-01):**

Add `_exec` with a `spawn` method. lifecycle.ts uses `Bun.spawn` not `$`, so the shape differs:

```typescript
export const _exec = {
  spawn: async (args: {
    cmd: string[]
    cwd: string
    env: Record<string, string>
    stdout: "inherit" | "pipe"
    stderr: "inherit" | "pipe"
  }): Promise<{
    exitCode: number
    stdout: ReadableStream<Uint8Array> | null
    stderr: ReadableStream<Uint8Array> | null
  }> => {
    const proc = spawn(args.cmd, {
      cwd: args.cwd,
      env: args.env,
      stdout: args.stdout,
      stderr: args.stderr,
    })
    const exitCode = await proc.exited
    return {
      exitCode,
      stdout: args.stdout === "pipe" ? proc.stdout : null,
      stderr: args.stderr === "pipe" ? proc.stderr : null,
    }
  },
}
```

Wait — the current code reads stdout/stderr streams BEFORE awaiting `proc.exited` (it uses `Promise.all` with stream readers). The `_exec.spawn` must NOT await `proc.exited` internally. Instead, return a handle that the caller awaits:

```typescript
export type SpawnHandle = {
  exited: Promise<number>
  stdout: ReadableStream<Uint8Array> | null
  stderr: ReadableStream<Uint8Array> | null
}

export const _exec = {
  spawn: (args: {
    cmd: string[]
    cwd: string
    env: Record<string, string>
    stdout: "inherit" | "pipe"
    stderr: "inherit" | "pipe"
  }): SpawnHandle => {
    const proc = spawn(args.cmd, {
      cwd: args.cwd,
      env: args.env,
      stdout: args.stdout,
      stderr: args.stderr,
    })
    return {
      exited: proc.exited,
      stdout: args.stdout === "pipe" ? proc.stdout : null,
      stderr: args.stderr === "pipe" ? proc.stderr : null,
    }
  },
}
```

Then in `runHooks`:
```typescript
const handle = _exec.spawn({
```
