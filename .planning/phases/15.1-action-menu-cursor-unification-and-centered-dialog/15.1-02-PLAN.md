export function ConfirmDialog(props: Props) {

        <text fg="yellow">  {props.message}</text>

  <action>
Restructure App.tsx to promote centered dialog views from the bottom detail pane box to root-level Show blocks. This prevents Pitfall 1 (dialog clipped to detail pane height) per RESEARCH.md.

**Key principle:** Dialogs render AFTER the split-pane Show block in the component tree. OpenTUI's z-ordering respects DOM order — later children paint over earlier children. The split pane remains visible underneath; the CenteredDialog's dim layer covers it visually.

**Step 1: Restructure the JSX return.** The current structure is:
```
<box> (root)
  <Show when={helpOpen()}> HelpOverlay </Show>
  <Show when={!helpOpen() && messagesOpen()}> MessageOverlay </Show>
  <Show when={!helpOpen() && !messagesOpen()}>
    <box> (top: lists) </box>
    <box> (bottom: detail + action menus + confirm + inline-input + ...) </box>
    <box> (help bar) </box>
  </Show>
</box>
```

Change to:
```
<box> (root)
  <Show when={helpOpen()}> HelpOverlay </Show>
  <Show when={!helpOpen() && messagesOpen()}> MessageOverlay </Show>
  <Show when={!helpOpen() && !messagesOpen()}>
    <box> (top: lists) </box>
    <box> (bottom: ONLY detail views for list state) </box>
    <box> (help bar) </box>
  </Show>

  {/* Promoted centered dialogs — render AFTER split pane, on top via DOM order */}
  <Show when={!helpOpen() && !messagesOpen() && view().view === "action-menu"}>
    {/* ActionMenu Switch for workspace/template tabs */}
  </Show>
  <Show when={!helpOpen() && !messagesOpen() && view().view === "confirm"}>
    {/* ConfirmDialog with title */}
  </Show>
  <Show when={!helpOpen() && !messagesOpen() && view().view === "inline-input"}>
    {/* InlineInput */}
  </Show>
  <Show when={!helpOpen() && !messagesOpen() && view().view === "repo-action-menu"}>
    {/* RepoActionMenu */}
  </Show>
  <Show when={!helpOpen() && !messagesOpen() && view().view === "repo-remove-blocked"}>
    {/* RemoveBlockedView */}
  </Show>
</box>
```
