```ts
function formatPushRow(row: PushRow): string {
  const label = row.status === "pushed" ? "pushed" : row.status === "skipped" ? "skipped" : "failed"
  return `${label}  ${row.repo}  ${row.detail}`
}
```

Register the command:

```ts
program
  .command("push [workspace]")
  .description("Push workspace branches to remote")
  .option("--force-with-lease", "Safe force-push (fails if remote has unseen commits)")
  .option("--force", "Hard force-push (prefer --force-with-lease)")
  .option("--dry-run", "Show what would be pushed without executing")
  .option("--set-upstream", "Set upstream tracking on push (-u)")
  .option("--json", "Output results as JSON")
  .action(async (workspace: string | undefined, opts: { forceWithLease?: boolean; force?: boolean; dryRun?: boolean; setUpstream?: boolean; json?: boolean }) => {
    let name = workspace

    // CWD detection fallback (same pattern as sync/status)
    if (!name) {
      const detected = detectWorkspaceFromCwd()
      if (detected) {
        name = detected
      }
    }

    if (opts.json) {
      if (!name) {
        console.log(JSON.stringify({ ok: false, error: "No workspace specified and could not detect from CWD" }))
        process.exit(1)
      }

      const result = await pushWorkspace(name, {
        force: opts.force,
        forceWithLease: opts.forceWithLease,
        dryRun: opts.dryRun,
        setUpstream: opts.setUpstream,
      })

      console.log(JSON.stringify({
        workspace: name,
        repos: [
          ...result.pushed.map(p => ({ name: p.repo, status: "pushed", commits: p.commits })),
          ...result.skipped.map(s => ({ name: s.repo, status: "skipped", reason: s.reason })),
          ...result.failed.map(f => ({ name: f.repo, status: "failed", reason: f.reason })),
        ],
        ok: result.ok,
        ...(result.error ? { error: result.error } : {}),
      }))

      if (!result.ok) process.exit(1)
      return
    }

    if (!name) {
      console.error(formatError("Missing workspace name", "usage: git-stacks push <name> [--force-with-lease] [--force] [--dry-run] [--json]"))
      process.exit(1)
    }

    if (!workspaceExists(name)) {
      console.error(formatError(`Workspace '${name}' not found`, "run: git-stacks list"))
      process.exit(1)
    }

    const result = await pushWorkspace(name, {
      force: opts.force,
      forceWithLease: opts.forceWithLease,
      dryRun: opts.dryRun,
      setUpstream: opts.setUpstream,
    }, (row) => console.log(`  ${formatPushRow(row)}`))

    if (!result.ok) {
      if (result.error) console.error(formatError(result.error))
      process.exit(1)
    }

    if (result.pushed.length === 0 && result.skipped.length === 0 && result.failed.length === 0) {
      console.log("Nothing to push.")
    }
  })
```

Also import `PushRow` type from `../lib/workspace-ops` alongside the existing imports and add `detectWorkspaceFromCwd` if not already imported (it should already be imported for sync/status).
