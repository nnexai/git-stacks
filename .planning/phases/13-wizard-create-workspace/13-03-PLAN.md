**Step 4: Build ad-hoc wizard step definitions**

Add a helper function near `buildTemplateWizardSteps`:
```typescript
function buildAdhocWizardSteps(repoNames: string[]): WizardStep<CreateWizardData>[] {
  return [
    {
      kind: "text",
      label: "Workspace name",
      key: "name",
      validate: (v: string) => {
        if (!v.trim()) return "Required"
        if (workspaceExists(v.trim())) return `Workspace '${v.trim()}' already exists`
        return undefined
      },
    },
    {
      kind: "text",
      label: "Branch",
      key: "branch",
      validate: (v: string) => (v.trim() ? undefined : "Required"),
    },
    {
      kind: "confirm",
      buildMessage: (data: Partial<CreateWizardData>) => {
        const repoLines = repoNames.map(n => `  ${n} (worktree)`).join("\n")
        return `Ad-hoc workspace\nBranch: ${data.branch}\nRepos:\n${repoLines}`
      },
    },
  ]
}
```

Note the differences from template wizard: no prefill on branch (per D-08), summary says "Ad-hoc workspace" instead of template name.

**Step 5: Add wizard-create-adhoc rendering in detail box JSX**
