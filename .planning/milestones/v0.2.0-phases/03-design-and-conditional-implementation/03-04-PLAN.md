    try {

The updated function should look like:
```typescript
export function warnExternalFiles(
  workspace: Workspace,
  wsDir: string,
  tasksDir: string,
): string[] {
  const warnings: string[] = []
  const allEntries: string[] = []

  // Collect from workspace-level files
  if (workspace.files?.copy) allEntries.push(...workspace.files.copy)
  if (workspace.files?.symlink) allEntries.push(...workspace.files.symlink)

  // Collect from per-repo files
  for (const repo of workspace.repos) {
    if (repo.files?.copy) allEntries.push(...repo.files.copy)
    if (repo.files?.symlink) allEntries.push(...repo.files.symlink)
  }

  for (const entry of allEntries) {
    const expanded = expandHome(entry)
    if (isAbsolute(expanded) && !expanded.startsWith(wsDir) && !expanded.startsWith(tasksDir)) {
      warnings.push(`\u26A0 external file destination will not be cleaned up: ${expanded}`)
    }
  }

  return warnings
}
```
