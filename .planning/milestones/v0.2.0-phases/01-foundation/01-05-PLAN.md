```typescript
async function setupWorkspaceFixture(tmp: string, opts: { repoCount?: number } = {}) {
  process.env.HOME = tmp
  const repoCount = opts.repoCount ?? 1

  // Dynamic imports for config isolation
  const config = await import("../../src/lib/config")
  const ops = await import("../../src/lib/workspace-ops")
  const git = await import("../../src/lib/git")
  const paths = await import("../../src/lib/paths")

  // Set up workspace root structure
  const wsRoot = join(tmp, "workspaces")
  const tasksDir = join(wsRoot, "tasks")
  const mainDir = join(wsRoot, "main")

  // Write global config
  config.writeGlobalConfig({ workspace_root: wsRoot, integrations: {} })

  // Create real git repos
  const repos: Array<{ name: string; repoPath: string; worktreePath: string }> = []
  for (let i = 0; i < repoCount; i++) {
    const name = `repo-${i}`
    const repoPath = makeGitRepo(mainDir, name)
    const worktreePath = join(tasksDir, "test-ws", name)
    repos.push({ name, repoPath, worktreePath })
  }

  // Write stack YAML
  config.writeStack({
    name: "test-stack",
    repos: repos.map((r) => ({
      name: r.name,
      path: r.repoPath,
      type: "other" as const,
      default_mode: "worktree" as const,
      default_branch: "main",
    })),
  })

  // Write workspace YAML
  const wsRepos = repos.map((r) => ({
    name: r.name,
    stack: "test-stack",
    type: "other" as const,
    mode: "worktree" as const,
    main_path: r.repoPath,
    task_path: r.worktreePath,
  }))

  config.writeWorkspace({
    name: "test-ws",
    branch: "feature/test",
    created: new Date().toISOString(),
    repos: wsRepos,
  })

  // Create actual worktrees
  for (const repo of repos) {
    await git.createWorktree(repo.repoPath, repo.worktreePath, "feature/test")
  }

  return { config, ops, git, paths, repos, wsRoot, tasksDir }
}
```

  Call `ops.mergeWorkspace("test-ws", { force: true })`.
  Assert: `result.ok === true`, workspace YAML no longer exists (`config.workspaceExists("test-ws") === false`).

- test: "preserves workspace YAML when merge fails (BUG-01)" --
  Set up with 2 repos. Add a commit to repo-0's worktree on feature/test.
  For repo-1, create a conflict: add a commit on main AND on feature/test that modify the same file. The conflict pre-check will catch this and return an error without attempting merges.
  Call `ops.mergeWorkspace("test-ws", { force: true })`.
  Assert: `result.ok === false`, `result.error` contains a string, workspace YAML STILL EXISTS (`config.workspaceExists("test-ws") === true`).

**describe("removeWorkspace")** (BUG-02):

- test: "removes worktrees and deletes workspace YAML" --
  Set up with 1 repo. Call `ops.removeWorkspace("test-ws", { force: true })`.
  Assert: `result.ok === true`, workspace YAML no longer exists, worktree directory no longer exists.

- test: "preserves YAML when worktree removal fails (BUG-02)" --
  Set up with 2 repos. Delete one worktree directory manually but leave git registration stale (or make the worktree removal fail some other way). Actually, the simpler approach: lock a file in the worktree so git worktree remove fails. Or: delete the main repo's .git so `git worktree remove` fails.
  Alternative approach: After setup, corrupt repo-1's main .git directory so `removeWorktree` throws for that repo.
  Call `ops.removeWorkspace("test-ws", { force: true })`.
  Assert: `result.ok === false`, workspace YAML STILL EXISTS.

**describe("cleanWorkspace")** (BUG-02):

- test: "removes worktrees without deleting workspace YAML" --
  Set up with 1 repo. Call `ops.cleanWorkspace("test-ws", { force: true })`.
  Assert: `result.ok === true`, workspace YAML still exists, worktree directory no longer exists.

**describe("renameWorkspace")** (BUG-03):

- test: "re-registers worktrees at new paths" --
  Set up with 1 repo. Call `ops.renameWorkspace("test-ws", "renamed-ws")`.
  Assert: `result.ok === true`.

  Old workspace YAML does not exist.
  The new worktree path exists on disk.
  The old worktree path does NOT exist on disk.

  The old worktree is NOT registered: `git.isWorktreeRegistered(repo.repoPath, oldWorktreePath)` returns false.

- test: "only re-registers worktree-mode repos, not trunk" --

  Call `ops.renameWorkspace("test-ws", "renamed-ws")`.
  Assert: `result.ok === true`. Trunk repo's main_path is unchanged.
