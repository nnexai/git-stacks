1. **Add ahead/behind computation memos** after the existing `countsText` memo (around line 38). The data comes from `props.entry.status`:

```typescript
  // Ahead/behind indicators (D-07: after branch, before counts)
  const aheadBehindText = createMemo(() => {
    const s = props.entry.status
    if (s.state !== "loaded") return ""
    const stale = s.aheadBehindStale ? "?" : ""
    // Aggregate: sum ahead, max behind from per-repo status
    let totalAhead = 0
    let maxBehind = 0
    for (const r of s.repos) {
      if (r.mode === "worktree") {
        totalAhead += r.ahead
        if (r.behind > maxBehind) maxBehind = r.behind
      }
    }
    const parts: string[] = []
    if (totalAhead > 0) parts.push(`↑${totalAhead}${stale}`)
    if (maxBehind > 0) parts.push(`↓${maxBehind}${stale}`)
    return parts.join(" ")
  })

  const aheadCount = createMemo(() => {
    const s = props.entry.status
    if (s.state !== "loaded") return 0
    return s.repos.filter(r => r.mode === "worktree").reduce((sum, r) => sum + r.ahead, 0)
  })

  const behindCount = createMemo(() => {
    const s = props.entry.status
    if (s.state !== "loaded") return 0
    return Math.max(0, ...s.repos.filter(r => r.mode === "worktree").map(r => r.behind))
  })

  const isStale = createMemo(() => {
    const s = props.entry.status
    return s.state === "loaded" && s.aheadBehindStale
  })
```
