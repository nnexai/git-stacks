Create new file `src/lib/labels.ts` with:

```ts
import type { Workspace } from "./config"

/**
 * Check if a workspace matches ALL given label terms (AND logic).
 * Returns true if workspace has every label in `terms`.
 * Case-sensitive comparison per D-13 in CONTEXT.md.
 * Empty terms array returns true (no filter = match all).
 */
export function matchesLabels(workspace: Workspace, terms: string[]): boolean {
  if (terms.length === 0) return true
  const wsLabels = workspace.labels ?? []
  return terms.every(term => wsLabels.includes(term))
}
```

This is the shared utility referenced by STATE.md constraint "LBL-01 label filter must use shared `matchesLabels(workspace, terms[])` utility".
</action>

```ts
import { describe, test, expect } from "bun:test"
import { matchesLabels } from "../../src/lib/labels"
import { WorkspaceSchema } from "../../src/lib/config"

function makeWs(labels?: string[]) {
  return WorkspaceSchema.parse({
    name: "test-ws",
    branch: "feature/test",
    created: "2026-01-01",
    labels,
  })
}

describe("matchesLabels", () => {
  test("returns true when terms is empty", () => {
    expect(matchesLabels(makeWs(["backend"]), [])).toBe(true)
  })

  test("returns true when workspace has no labels and terms is empty", () => {
    expect(matchesLabels(makeWs(), [])).toBe(true)
  })

  test("returns false when workspace has no labels but terms are given", () => {
    expect(matchesLabels(makeWs(), ["backend"])).toBe(false)
  })

  test("matches single label", () => {
    expect(matchesLabels(makeWs(["backend", "sprint:14"]), ["backend"])).toBe(true)
  })

  test("AND logic — all terms must match", () => {
    expect(matchesLabels(makeWs(["backend", "sprint:14"]), ["backend", "sprint:14"])).toBe(true)
    expect(matchesLabels(makeWs(["backend"]), ["backend", "sprint:14"])).toBe(false)
  })

  test("case-sensitive matching", () => {
    expect(matchesLabels(makeWs(["Backend"]), ["backend"])).toBe(false)
    expect(matchesLabels(makeWs(["backend"]), ["Backend"])).toBe(false)
  })

  test("namespaced labels with colons", () => {
    expect(matchesLabels(makeWs(["sprint:14", "client:acme"]), ["sprint:14"])).toBe(true)
  })
})
```
</action>
