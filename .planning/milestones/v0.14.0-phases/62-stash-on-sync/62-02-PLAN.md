  // --- Stash Phase 3: Pop stashes in reverse ---
  const stashPopFailures: Array<{ repo: string; error: string; repoPath: string }> = []
  if (opts.stash && stashedRepos.length > 0) {
    for (const repo of [...stashedRepos].reverse()) {
      onProgress?.({ repo: repo.name, status: "popping", detail: "", conflicts: [] })
      const popResult = await stashPop(repo.task_path)
      if (!popResult.ok) {
        stashPopFailures.push({ repo: repo.name, error: popResult.error, repoPath: repo.task_path })
        onProgress?.({ repo: repo.name, status: "failed", detail: `stash pop failed`, conflicts: [] })

      }

  return {

  async function executeSync(name: string) {
    const ws = readWorkspace(name)
    const worktreeRepos = ws.repos.filter(r => r.mode === "worktree")
    const initialRows: SyncRow[] = worktreeRepos
      .map(r => ({ repo: r.name, status: "pending" as const, detail: "", conflicts: [] }))

    setSyncRows(initialRows)
    setSyncDone(false)
    setSyncSummary({ text: "", color: "green" })
    setView({ view: "sync-progress", message: `Syncing ${name}...` })

    // Check if any repos are dirty ‚Äî auto-stash if so (D-01)
    let hasDirty = false
    for (const repo of worktreeRepos) {
      if (existsSync(repo.task_path) && await isRepoDirty(repo.task_path)) {
        hasDirty = true
        break
      }
    }

    const onProgress = (update: SyncRow) => {
      setSyncRows(prev => prev.map(r => r.repo === update.repo ? { ...r, ...update } : r))
    }

    try {
      const result = await syncWorkspace(name, { strategy: "rebase", bestEffort: true, stash: hasDirty }, onProgress)
      setSyncSummary(buildSummary(result))
    } catch (e) {
      const msg = e instanceof Error ? e.message : String(e)
      setSyncSummary({ text: `Sync failed: ${msg}. Press any key to continue.`, color: "red" })
    }
    setSyncDone(true)
  }

  function buildSummary(result: SyncResult): { text: string; color: "green" | "yellow" | "red" } {
    const ns = result.synced.length
    const nsk = result.skipped.length
    const nf = result.stashPopFailures?.length ?? 0

    if (ns === 0 && nsk === 0 && nf === 0) {
      return { text: "Nothing to sync. Press any key to continue.", color: "green" }
    }
    if (nf > 0) {
      const repos = result.stashPopFailures!.map(f => f.repo).join(", ")
      return { text: `${ns} synced. Stash pop conflict in: ${repos}. Press any key to continue.`, color: "red" }
    }
    if (nsk > 0) {
      return { text: `${ns} synced, ${nsk} skipped. Press any key to continue.`, color: "yellow" }
    }
    return { text: `${ns} synced. Press any key to continue.`, color: "green" }
  }

  beforeEach(() => {
    tmp = makeTmpDir("sync-stash-test")
    process.env.GIT_STACKS_CONFIG_DIR = join(tmp, ".config")

  /**
   * Setup: bare origin ‚Üí main clone ‚Üí worktree.
   * Returns { originPath, mainPath, taskPath, wsName }.
   */
  async function setupSyncFixture(name: string) {
    const originPath = join(tmp, `${name}-origin`)
    const mainPath = join(tmp, `${name}-main`)
    const taskPath = join(tmp, `${name}-task`)

    // Create bare origin
    execSync(`git init --bare ${originPath}`)

    // Clone into main
    execSync(`git clone ${originPath} ${mainPath}`)
    execSync(`git -C ${mainPath} checkout -b main`, { stdio: "pipe" })
    writeFileSync(join(mainPath, "README.md"), "initial")
    execSync(`git -C ${mainPath} add . && git -C ${mainPath} commit -m "init"`, { stdio: "pipe" })
    execSync(`git -C ${mainPath} push -u origin main`, { stdio: "pipe" })

    // Create worktree branch and push
    execSync(`git -C ${mainPath} checkout -b feature/${name}`, { stdio: "pipe" })
    execSync(`git -C ${mainPath} push -u origin feature/${name}`, { stdio: "pipe" })

    // Create worktree
    execSync(`git -C ${mainPath} worktree add ${taskPath} feature/${name}`, { stdio: "pipe" })

    // Write registry
    const configDir = join(tmp, ".config")
    mkdirSync(join(configDir, "workspaces"), { recursive: true })
    writeFileSync(join(configDir, "registry.yml"), `- name: ${name}-repo\n  path: ${mainPath}\n  type: generic\n  default_branch: main\n`)

    // Write workspace YAML
    const wsYaml = `name: ${name}\nbranch: feature/${name}\ncreated: "2026-01-01"\nrepos:\n  - name: ${name}-repo\n    repo: ${name}-repo\n    mode: worktree\n    base_branch: main\n    main_path: ${mainPath}\n    task_path: ${taskPath}\n`
    writeFileSync(join(configDir, `workspaces/${name}.yml`), wsYaml)

    return { originPath, mainPath, taskPath, wsName: name }
