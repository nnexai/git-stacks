```ts
/**
 * Stash uncommitted changes (tracked + untracked) with a marker message.
 * Returns ok:true with stashRef on success, ok:false with error on failure.
 */
export async function stashPush(
  repoPath: string,
  message: string
): Promise<{ ok: true; stashRef: string } | { ok: false; error: string }> {
  const result = await $`git -C ${repoPath} stash push --include-untracked -m ${message}`.quiet().nothrow()
  if (result.exitCode !== 0) {
    return { ok: false, error: result.stderr.toString().trim() || "stash push failed" }
  }
  // git stash push outputs "Saved working directory and index state ..." on success
  // but outputs "No local changes to save" (exit 0) when nothing to stash
  const stdout = result.stdout.toString().trim()
  if (stdout.includes("No local changes to save")) {
    return { ok: false, error: "nothing to stash" }
  }
  // Resolve the stash ref — always stash@{0} after a successful push
  return { ok: true, stashRef: "stash@{0}" }
}
```

Key design choices:
- Uses `--include-untracked` per D-07 to capture new files
- Uses `.quiet().nothrow()` per established pattern
- Handles the edge case where git returns exit 0 but nothing was stashed
- Returns discriminated union matching project convention (`{ ok: true } | { ok: false; error }`)
</action>

/**
 * Pop the top stash entry. Detects conflicts from exit code + stderr.
 * On conflict: stash is preserved (git does not drop it), returns conflict: true.
 */
export async function stashPop(
  repoPath: string
): Promise<{ ok: true } | { ok: false; conflict: boolean; error: string }> {
  const result = await $`git -C ${repoPath} stash pop`.quiet().nothrow()
  if (result.exitCode === 0) {
    return { ok: true }
  }
  const stderr = result.stderr.toString()
  const stdout = result.stdout.toString()
  const conflict = stderr.includes("CONFLICT") || stdout.includes("CONFLICT")
  return { ok: false, conflict, error: stderr.trim() || stdout.trim() || "stash pop failed" }
}
```

Key design choices:
- When `git stash pop` fails due to conflict, git preserves the stash entry (does not drop it) — this is standard git behavior and matches D-04
- Conflict detection checks both stderr and stdout since git versions vary in where they output CONFLICT markers
- Returns `conflict: boolean` so callers can distinguish merge conflicts from other errors
</action>

/**
 * Check if repo has an existing git-stacks auto-stash entry.
 * Used as a double-stash guard — if found, sync should refuse to stash again.
 */
export async function hasAutoStash(repoPath: string): Promise<boolean> {
  const result = await $`git -C ${repoPath} stash list`.quiet().nothrow()
  if (result.exitCode !== 0) return false
  return result.stdout.toString().includes("git-stacks auto-stash")
}
```

Key design choices:
- Per D-02/D-03: grep `git stash list` output for the marker string `git-stacks auto-stash`
- This is reliable because the marker is set by our `stashPush` call with the specific message
- Returns `false` on any error (conservative — don't block sync due to stash list failure)
</action>

// ---------------------------------------------------------------------------
// describe("stash primitives") — STH-01, STH-02

describe("stash primitives", () => {
  let tmp: string
  let repoPath: string

  beforeEach(() => {
    tmp = makeTmpDir("stash-test")
    repoPath = makeGitRepo(tmp)
  })
  afterEach(() => cleanup(tmp))

  test("stashPush stashes dirty tracked file", async () => {
    writeFileSync(join(repoPath, "README.md"), "modified content")
    const result = await stashPush(repoPath, "git-stacks auto-stash (sync)")
    expect(result.ok).toBe(true)
    if (result.ok) {
      expect(result.stashRef).toBe("stash@{0}")
    }
    // Verify working tree is clean after stash
    const status = execSync(`git -C ${repoPath} status --porcelain`, { encoding: "utf-8" }).trim()
    expect(status).toBe("")
  })

  test("stashPush stashes untracked file", async () => {
    writeFileSync(join(repoPath, "new-file.txt"), "new content")
    const result = await stashPush(repoPath, "git-stacks auto-stash (sync)")
    expect(result.ok).toBe(true)
    // Verify untracked file is gone after stash
    const status = execSync(`git -C ${repoPath} status --porcelain`, { encoding: "utf-8" }).trim()
    expect(status).toBe("")
  })

  test("stashPush returns error when nothing to stash", async () => {
    const result = await stashPush(repoPath, "git-stacks auto-stash (sync)")
    expect(result.ok).toBe(false)
    if (!result.ok) {
      expect(result.error).toContain("nothing to stash")
    }
  })

  test("stashPop restores stashed changes", async () => {
    writeFileSync(join(repoPath, "README.md"), "modified content")
    await stashPush(repoPath, "git-stacks auto-stash (sync)")
    const result = await stashPop(repoPath)
    expect(result.ok).toBe(true)
    // Verify file is restored
    const content = execSync(`cat ${join(repoPath, "README.md")}`, { encoding: "utf-8" }).trim()
    expect(content).toBe("modified content")
  })

  test("stashPop returns ok:false when no stash exists", async () => {
    const result = await stashPop(repoPath)
    expect(result.ok).toBe(false)
    if (!result.ok) {
      expect(result.error).toBeTruthy()
    }
  })

  test("hasAutoStash returns false on clean repo", async () => {

  test("hasAutoStash returns true when auto-stash entry exists", async () => {
    writeFileSync(join(repoPath, "README.md"), "dirty")

    expect(result).toBe(false)
  })
})
