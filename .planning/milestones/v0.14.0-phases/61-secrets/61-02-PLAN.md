### Task 2: Add buildResolvers function to secrets.ts

<read_first>
- src/lib/secrets.ts (SecretResolver interface, resolver instances from Plan 01)
- src/lib/config.ts (GlobalConfig type with new secrets field)
</read_first>

<action>
Add `buildResolvers` to `src/lib/secrets.ts`. This function reads the global config's `secrets.resolvers` list and returns the matching resolver instances.

```ts
import type { GlobalConfig } from "./config"

/** Default enabled resolvers when config.secrets.resolvers is not set */
const DEFAULT_RESOLVERS = ["keychain", "env"]

/** All built-in resolvers keyed by id */
const RESOLVER_REGISTRY: Record<string, SecretResolver> = {
  keychain: keychainResolver,
  env: envResolver,
  cmd: cmdResolver,
}

export function buildResolvers(config: GlobalConfig): SecretResolver[] {
  const enabledIds = config.secrets?.resolvers ?? DEFAULT_RESOLVERS
  const resolvers: SecretResolver[] = []
  for (const id of enabledIds) {
    const resolver = RESOLVER_REGISTRY[id]
    if (resolver) resolvers.push(resolver)
    // Unknown ids are silently ignored — future resolvers may not exist in this version
  }
  return resolvers
}
```

Key behaviors:
- When `config.secrets` is undefined (no config): returns `[keychainResolver, envResolver]`
- When `config.secrets.resolvers` is undefined: same default
- When `config.secrets.resolvers` is `["keychain", "env", "cmd"]`: returns all three
- When `config.secrets.resolvers` is `["env"]`: returns only envResolver
- `cmd` resolver is NOT in the default list — requires explicit opt-in (D-04, SEC-05)
</action>

<acceptance_criteria>
- `buildResolvers({ workspace_root: "/tmp", integrations: {}, ports: { range_start: 10000, range_end: 65000 } })` returns array with 2 resolvers (keychain, env) — cmd is excluded by default
- `buildResolvers({ ...defaults, secrets: { resolvers: ["keychain", "env", "cmd"] } })` returns array with 3 resolvers
- `buildResolvers({ ...defaults, secrets: { resolvers: ["env"] } })` returns array with 1 resolver (env only)
- `buildResolvers({ ...defaults, secrets: { resolvers: [] } })` returns empty array
- `buildResolvers({ ...defaults, secrets: {} })` returns default resolvers (keychain, env)
- `src/lib/secrets.ts` exports `buildResolvers`
</acceptance_criteria>

```ts
  test("parses config without secrets field (backward compat)", () => {

    expect(config.secrets).toBeUndefined()
  })

  test("parses config with secrets.resolvers", () => {
    const config = GlobalConfigSchema.parse({ secrets: { resolvers: ["keychain", "env"] } })
    expect(config.secrets?.resolvers).toEqual(["keychain", "env"])

  test("parses config with empty secrets object", () => {
    const config = GlobalConfigSchema.parse({ secrets: {} })
    expect(config.secrets).toBeDefined()
    expect(config.secrets?.resolvers).toBeUndefined()
  })

  test("rejects non-string resolver entries", () => {
    expect(() => GlobalConfigSchema.parse({ secrets: { resolvers: [123] } })).toThrow()
  })
})
```
