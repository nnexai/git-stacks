## Objective

Wire `resolveSecrets` into `openWorkspace()`, add the `--skip-secrets` CLI flag to the `open` command, and add a "Secrets" section to the config wizard. This plan depends on Plans 01 and 02 being complete.

2. Add `skipSecrets?: boolean` to the `openWorkspace` opts parameter. Change:

To:
```ts

  onProgress?: ProgressCallback
): Promise<{ ok: boolean; error?: string }> {
```

3. Replace the env block (lines 842-844). Change:
```ts
  const mergedEnvVars = mergeEnv(wsWithPorts)
  writeEnvFiles(wsWithPorts, mergedEnvVars, msg => onProgress?.(msg))
  const hookEnv = { ...baseEnv, ...mergedEnvVars }
```
To:
```ts
  const mergedEnvVars = mergeEnv(wsWithPorts)
  
  // Secret resolution — resolve ${{ resolver:path }} references before writing env files
  let resolvedEnvVars: Record<string, string>
  if (opts.skipSecrets) {
    // --skip-secrets: substitute empty strings for unresolved references, log warnings
    resolvedEnvVars = {}
    for (const [key, value] of Object.entries(mergedEnvVars)) {
      const ref = parseSecretRef(value)
      if (ref) {
        onProgress?.(`\u26A0 Skipping secret: ${key} (${ref.id}:${ref.path})`)
        resolvedEnvVars[key] = ""
      } else {
        resolvedEnvVars[key] = value
      }
    }
  } else {
    try {
      const resolvers = buildResolvers(config)
      resolvedEnvVars = await resolveSecrets(mergedEnvVars, resolvers)
    } catch (err) {
      return { ok: false, error: `Secret resolution failed: ${err instanceof Error ? err.message : String(err)}` }
    }
  }
  
  writeEnvFiles(wsWithPorts, resolvedEnvVars, msg => onProgress?.(msg))
  const hookEnv = { ...baseEnv, ...resolvedEnvVars }
```

4. Add import for `parseSecretRef`:
```ts
import { resolveSecrets, buildResolvers, parseSecretRef } from "./secrets"
```

Critical invariant (SEC-08): The `wsWithPorts` object is NOT modified — it still contains the raw `${{ ... }}` references. Only `resolvedEnvVars` (a new local variable) contains resolved plaintext. The existing `writeWorkspace(wsWithPorts)` call at line 739 writes the pre-resolution workspace.
</action>

<read_first>

<action>

    // Secrets resolver configuration
    const allResolverIds = ["keychain", "env", "cmd"]
    const currentResolvers = config.secrets?.resolvers ?? ["keychain", "env"]
    
    const selectedResolversRaw = await p.multiselect({
      message: "Enabled secret resolvers",
      options: [
        { value: "keychain", label: "keychain", hint: "macOS Keychain / Linux secret-tool" },

      ],
      initialValues: currentResolvers,
      required: false,
    })
