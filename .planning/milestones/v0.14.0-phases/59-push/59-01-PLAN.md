```ts
export async function pushBranch(
  repoPath: string,
  branch: string,
  opts: { force?: boolean; forceWithLease?: boolean; setUpstream?: boolean } = {}
): Promise<{ ok: true; commits: number } | { ok: false; reason: string }> {
  // Count commits that will be pushed (ahead of remote)
  const countResult = await $`git -C ${repoPath} rev-list --count ${"origin/" + branch + "..HEAD"}`.quiet().nothrow()
  const commitsBefore = countResult.exitCode === 0 ? parseInt(countResult.stdout.toString().trim(), 10) || 0 : 0

  // Build push args
  const args: string[] = ["git", "-C", repoPath, "push"]
  if (opts.setUpstream) args.push("-u")
  if (opts.forceWithLease) args.push("--force-with-lease")
  else if (opts.force) args.push("--force")
  args.push("origin", branch)

  const result = await $`GIT_TERMINAL_PROMPT=0 ${args}`.quiet().nothrow()
  if (result.exitCode !== 0) {
    const stderr = result.stderr.toString()
    if (stderr.includes("non-fast-forward")) {
      return { ok: false, reason: "non-fast-forward (use --force-with-lease)" }
    }
    if (stderr.includes("does not match any")) {
      return { ok: false, reason: `no remote branch: ${branch}` }
    }
    if (stderr.includes("Authentication failed") || stderr.includes("could not read Username")) {
      return { ok: false, reason: "authentication failed" }
    }
    if (stderr.includes("Could not resolve host") || stderr.includes("unable to access")) {
      return { ok: false, reason: "network error" }
    }
    return { ok: false, reason: stderr.split("\n").filter(l => l.trim()).pop() || "push failed" }
  }

  return { ok: true, commits: commitsBefore }
}
```

Note: The `$` template literal with an array spreads the args correctly in Bun shell. However, if Bun's `$` does not support array interpolation cleanly, construct the command string manually:

```ts
const cmdParts = ["GIT_TERMINAL_PROMPT=0", "git", "-C", repoPath, "push"]
if (opts.setUpstream) cmdParts.push("-u")
if (opts.forceWithLease) cmdParts.push("--force-with-lease")
else if (opts.force) cmdParts.push("--force")
cmdParts.push("origin", branch)
```

Use the pattern from `pullFFOnly` where `$\`GIT_TERMINAL_PROMPT=0 git -C ${repoPath} pull ...\`` is called — build a similar template literal with conditional flag insertion. The executor should verify which Bun `$` pattern works best by reading the existing code.

Important: `--force-with-lease` and `--force` are mutually exclusive. If both are set, `forceWithLease` takes precedence (it's safer).

The commit count is computed by counting commits ahead of `origin/<branch>` before the push. If the remote tracking ref doesn't exist (first push), `rev-list` will fail and commits defaults to 0.
