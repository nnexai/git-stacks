Create `src/tui/dashboard/PushProgressView.tsx` by adapting `SyncProgressView.tsx` for push status values:

```tsx
/** @jsxImportSource @opentui/solid */
import "opentui-spinner/solid"
import { For, Show } from "solid-js"
import { CenteredDialog } from "./CenteredDialog"

export type PushRowDisplay = {
  repo: string
  status: "pending" | "pushing" | "pushed" | "skipped" | "failed"
  detail: string
}

function glyphFor(status: PushRowDisplay["status"]): string {
  if (status === "pending") return "\u00b7"
  if (status === "pushed") return "\u2713"
  if (status === "skipped") return "\u26a0"
  if (status === "failed") return "\u2717"
  return "" // pushing uses spinner
}

function colorFor(status: PushRowDisplay["status"]): string {
  if (status === "pending") return "gray"
  if (status === "pushed") return "green"
  if (status === "skipped") return "yellow"
  if (status === "failed") return "red"
  return "cyan" // pushing
}

type Props = {
  rows: PushRowDisplay[]
  done: boolean
  summary: { text: string; color: "green" | "yellow" | "red" }
  title?: string
}

export function PushProgressView(props: Props) {
  return (
    <CenteredDialog title={props.title ?? "Push Progress"} size="medium">
      <box flexDirection="column">
      <Show when={!props.done}>
        <box flexDirection="row" height={1}>
          <spinner name="dots" color="cyan" />
          <text fg="white"> Pushing...</text>
        </box>
      </Show>
      <For each={props.rows}>
        {(row) => (
          <box flexDirection="row" height={1} paddingLeft={2}>
            <Show
              when={row.status === "pushing"}
              fallback={
                <text fg={colorFor(row.status)}>{glyphFor(row.status)} </text>
              }
            >
              <spinner name="dots" color="cyan" />
              <text fg="white"> </text>
            </Show>
            <text fg="white">{row.repo}</text>
            <text fg="gray">  {row.detail}</text>
          </box>
        )}
      </For>
      <Show when={props.done && !!props.summary.text}>
        <text fg={props.summary.color}>{"\n"}  {props.summary.text}</text>
      </Show>
      </box>
    </CenteredDialog>
  )
}
```

Key differences from SyncProgressView:
- Status values are push-specific: `"pushing"` instead of `"fetching"`/`"rebasing"`
- No `conflicts` array (push doesn't have merge conflicts)
- Spinner shows during `"pushing"` state only (sync has both `"fetching"` and `"rebasing"`)
- Title defaults to "Push Progress"
- Text shows "Pushing..." instead of "Syncing..."
</action>

```ts
async function executePush(name: string) {
  const ws = readWorkspace(name)
  const initialRows: PushRowDisplay[] = ws.repos
    .filter(r => r.mode === "worktree")
    .map(r => ({ repo: r.name, status: "pending" as const, detail: "" }))

  setPushRows(initialRows)
  setPushDone(false)
  setPushSummary({ text: "", color: "green" })
  setView({ view: "push-progress", message: `Pushing ${name}...` })

  const onProgress = (update: PushRow) => {
    setPushRows(prev => prev.map(r => r.repo === update.repo ? { ...r, ...update } : r))
  }

  try {
    const result = await pushWorkspace(name, {}, onProgress)
    // Build summary
    const parts: string[] = []
    if (result.pushed.length > 0) parts.push(`${result.pushed.length} pushed`)
    if (result.failed.length > 0) parts.push(`${result.failed.length} failed`)
    if (result.skipped.length > 0) parts.push(`${result.skipped.length} skipped`)
    const color = result.failed.length > 0 ? "red" as const : "green" as const
    const text = parts.length > 0
      ? `${parts.join(", ")}. Press any key to continue.`
      : "Nothing to push. Press any key to continue."
    setPushSummary({ text, color })
  } catch (e) {
    const msg = e instanceof Error ? e.message : String(e)
    setPushSummary({ text: `Push failed: ${msg}. Press any key to continue.`, color: "red" })
  }
  setPushDone(true)
}
```
