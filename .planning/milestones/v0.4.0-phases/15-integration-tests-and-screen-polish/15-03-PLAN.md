2. Returns a result object

```typescript
mock.module("../../../src/lib/workspace-ops", () => ({
  openWorkspace: mock(async () => {}),
  cleanWorkspace: mock(async () => ({ ok: true })),
  removeWorkspace: mock(async () => ({ ok: true })),
  mergeWorkspace: mock(async () => ({ ok: true })),
  renameWorkspace: mock(async () => {}),
  syncWorkspace: mock(async (_ws: any, _cfg: any, _strat: any, onProgress?: any) => {
    if (onProgress) {
      onProgress({ repo: "sync-repo", status: "fetching" })
      onProgress({ repo: "sync-repo", status: "rebasing" })
      onProgress({ repo: "sync-repo", status: "done" })

    return { ok: true, results: [{ repo: "sync-repo", status: "synced" }] }
  }),
}))
```
